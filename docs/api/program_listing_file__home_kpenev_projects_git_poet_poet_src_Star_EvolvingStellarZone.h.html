        //__current_age_quantities if already present.

        ) const;

        double current_age() {return __current_age;}

        double min_interp_age() const;

        virtual void select_interpolation_region(double age) const;

        ~EvolvingStellarZone();
    };//End EvolvingStellarZone class.
