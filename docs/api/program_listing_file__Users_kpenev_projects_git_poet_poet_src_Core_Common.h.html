        std::valarray<double> valarray_from_iterator(ITERATOR_TYPE values,

        {
            std::valarray<double> result(size);
            double *dest=&result[0];
            for(; size>0; --size) {
                *dest = *values;
                ++dest;
                ++values;
            }
            return result;
        }
