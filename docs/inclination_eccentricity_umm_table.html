A table of curly U values as originally defined in Lai (2012)¶

m’=-2¶

\[\begin{split}\mathcal{U}_{-2,-2}&=&\sqrt{\frac{3\pi}{10}}\frac{1}{4}(1+cos\theta)^2\\ \mathcal{U}_{-1,-2}&=&-\sqrt{\frac{3\pi}{10}}\frac{1}{2} \sin\theta(1+\cos\theta)\\ \mathcal{U}_{0,-2}&=&\sqrt{\frac{\pi}{5}}\frac{3}{4}\sin^2\theta\\ \mathcal{U}_{1,-2}&=&-\sqrt{\frac{3\pi}{10}}\frac{1}{2} \sin\theta(1-\cos\theta)\\ \mathcal{U}_{2,-2}&=&\sqrt{\frac{3\pi}{10}}\frac{1}{4}(1-\cos\theta)^2\\\end{split}\]

m’=0¶

\[\begin{split}\mathcal{U}_{-2,0}&=&-\sqrt{\frac{6\pi}{5}}\frac{1}{4}\sin^2\theta\\ \mathcal{U}_{-1,0}&=&-\sqrt{\frac{6\pi}{5}}\frac{1}{2} \sin\theta\cos\theta\\ \mathcal{U}_{0,0}&=&-\sqrt{\frac{\pi}{5}}\frac{1}{2}(3\cos^2\theta-1)\\ \mathcal{U}_{1,0}&=&\sqrt{\frac{6\pi}{5}}\frac{1}{2} \sin\theta\cos\theta\\ \mathcal{U}_{2,0}&=&-\sqrt{\frac{6\pi}{5}}\frac{1}{4}\sin^2\theta\end{split}\]

m’=2¶

\[\begin{split}\mathcal{U}_{-2,2}&=&\sqrt{\frac{3\pi}{10}}\frac{1}{4}(1-\cos\theta)^2\\ \mathcal{U}_{-1,2}&=&\sqrt{\frac{3\pi}{10}}\frac{1}{2} \sin\theta(1-\cos\theta)\\ \mathcal{U}_{0,2}&=&\sqrt{\frac{\pi}{5}}\frac{3}{4}\sin^2\theta\\ \mathcal{U}_{1,2}&=&\sqrt{\frac{3\pi}{10}}\frac{1}{2} \sin\theta(1+\cos\theta)\\ \mathcal{U}_{2,2}&=&\sqrt{\frac{3\pi}{10}}\frac{1}{4}(1+\cos\theta)^2\\\end{split}\]
